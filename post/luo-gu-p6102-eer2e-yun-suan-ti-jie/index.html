我们先看要求什么：

在解决含有异或，与，或之类的问题时，非常常用的方法就是分位解决。

设 a[i] 化为二进制后从右至左第 j 位为 wei[i][j]。

对于每一位，比如 pos 位。为了方便，下记 wei[i][pos]=a[i]。设 aia_iai​ 中有 xxx 个 111，n−xn-xn−x 个 000。首先原式化为

∑i=1n∑j=1n∑k=1n∑l=1n((ai or aj)+(ak and al))mod  2\sum\limits_{i=1}^n\sum\limits_{j=1}^n\sum\limits_{k=1}^n\sum\limits_{l=1}^n ((a_i\ \text{or}\ a_j)+(a_k\ \text{and}\ a_l))\mod{2} i=1∑n​j=1∑n​k=1∑n​l=1∑n​((ai​ or aj​)+(ak​ and al​))mod2

因为 aia_iai​只有两个取值（000 或 111）。然而

0 xor 0=1 xor 1=00\ \text{xor}\ 0=1\ \text{xor}\ 1=0 0 xor 0=1 xor 1=0

0 xor 1=1 xor 0=10\ \text{xor}\ 1=1\ \text{xor}\ 0=1 0 xor 1=1 xor 0=1

经归纳总结后，即 a xor b=(a+b)mod  2a\ \text{xor}\ b=(a+b)\mod{2}a xor b=(a+b)mod2。

代码：

讲解较清楚，所以代码没有过多介绍了。

#include<bits/stdc++.h>
#define int long long
using namespace std;
const int N=500005,Q=33;
int n;
int wei[N][Q],cnt[Q];
int ans=0;
int pow(int x){
	int kkks=1;
	return kkks<<x;
}
int cal(int x){return x*x;}
signed main(){
	scanf("%lld",&n);
	for(int i=1;i<=n;i++){
		int x;
		scanf("%lld",&x);
		for(int j=1;j<Q;j++){
			wei[i][j]=(x&1);//处理wei数组，作用详见讲解
			if(wei[i][j]==1) cnt[j]++;//cnt[i]表示第i位有几个1
			x>>=1;
		}
	}
	for(int i=1;i<Q;i++){
		int x=cnt[i];
		int pos1=x*(n-x);pos1%=pow(Q-i);
		int pos2=cal(n)-cal(x)+n*x;pos2%=pow(Q-i);
		int pos=2*pos1*pos2;pos%=pow(Q-i);
      //套公式
		ans+=pos<<(i-1);ans%=pow(32);
      //注意ans也要取模哦！
	}
	printf("%lld\n",ans);
}
